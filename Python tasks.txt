def fib(n):
    if n <= 0:
        return "Invalid input"
    elif n == 1:
        return 0
    elif n == 2:
        return 1
    else:
        return fib(n - 1) + fib(n - 2)

print(fib(6))
print()

def palindrome(s):
    if len(s) <= 1:  
        return True
    elif s[0] != s[-1]: 
        return False
    else:
        return palindrome(s[1:-1])
